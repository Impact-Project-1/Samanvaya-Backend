create table public.customers (
  id uuid not null default auth.uid (),
  display_name text not null default ''::text,
  profile_pic text not null default ''::text,
  description text not null default ''::text,
  contact jsonb not null default '{}'::jsonb,
  identification text not null default ''::text,
  constraint customers_pkey primary key (id),
  constraint customers_contact_key unique (contact),
  constraint customers_id_fkey1 foreign KEY (id) references auth.users (id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;

create table public.categories (
  name text not null default '''''::text'::text,
  slug text not null,
  category_id integer generated by default as identity not null,
  constraint categories_pkey primary key (category_id),
  constraint categories_name_key unique (name),
  constraint categories_slug_key unique (slug),
  constraint categories_id_check check ((category_id > 0)),
  constraint categories_name_check check ((length(name) > 3)),
  constraint categories_slug_check check ((length(slug) < 500))
) TABLESPACE pg_default;

create table public.reviews (
  vendor_id uuid not null,
  user_id uuid null,
  rating integer null,
  comment text null,
  created_at timestamp without time zone null default now(),
  constraint reviews_pkey primary key (vendor_id),
  constraint reviews_vendor_id_user_id_key unique (vendor_id, user_id),
  constraint reviews_user_id_fkey1 foreign KEY (user_id) references auth.users (id) on update CASCADE on delete CASCADE,
  constraint reviews_vendor_id_fkey foreign KEY (vendor_id) references vendors (vendor_id) on update CASCADE on delete CASCADE,
  constraint reviews_rating_check check (
    (
      (rating >= 1)
      and (rating <= 5)
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_reviews_vendor on public.reviews using btree (vendor_id) TABLESPACE pg_default;

create trigger review_insert_trigger
after INSERT on reviews for EACH row
execute FUNCTION update_vendor_rating ();

create table public.vendor_categories (
  vendor_id uuid not null,
  category_id integer not null default 0,
  constraint vendor_categories_pkey primary key (vendor_id, category_id),
  constraint vendor_categories_category_id_fkey foreign KEY (category_id) references categories (category_id) on update CASCADE on delete set default,
  constraint vendor_categories_vendor_id_fkey foreign KEY (vendor_id) references vendors (vendor_id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;

create table public.vendors (
  vendor_id uuid not null default auth.uid (),
  business_name text not null default ''::text,
  about text null default ''::text,
  city text not null default ''::text,
  state text not null default ''::text,
  phone text not null default ''::text,
  whatsapp text not null default ''::text,
  website text not null default ''::text,
  price_range_high double precision not null default 0,
  price_range_low double precision not null default 0,
  created_at timestamp without time zone not null default now(),
  rating_avg real null default 0,
  rating_count integer null default 0,
  links text[] not null default '{}'::text[],
  is_featured boolean null default false,
  status text not null default 'pending'::text,
  constraint vendors_pkey primary key (vendor_id),
  constraint vendors_business_name_key unique (business_name),
  constraint vendors_vendor_id_fkey1 foreign KEY (vendor_id) references auth.users (id) on update CASCADE on delete CASCADE,
  constraint high_price_check check ((price_range_high >= price_range_low)),
  constraint vendors_rating_range_check check ((rating_avg < ((10)::numeric)::double precision)),
  constraint vendors_description_check check ((length(about) < 500)),
  constraint low_price_check check ((price_range_low >= (0)::double precision)),
  constraint vendors_business_name_check check ((length(business_name) > 2))
) TABLESPACE pg_default;

create index IF not exists idx_vendors_city on public.vendors using btree (city) TABLESPACE pg_default;

create table public.vendor_images (
  vendor_id uuid not null,
  image_url text not null,
  constraint vendor_images_pkey primary key (vendor_id),
  constraint vendor_images_vendor_id_fkey foreign KEY (vendor_id) references vendors (vendor_id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;

create table public.chats (
  chat_id uuid not null default gen_random_uuid(),
  customer_id uuid not null,
  vendor_id uuid not null,
  created_at timestamp without time zone null default now(),
  constraint chats_pkey primary key (chat_id),
  constraint chats_customer_id_fkey foreign KEY (customer_id) references customers (id) on update CASCADE on delete CASCADE,
  constraint chats_vendor_id_fkey foreign KEY (vendor_id) references vendors (vendor_id) on update CASCADE on delete CASCADE,
  constraint chats_customer_vendor_key unique (customer_id, vendor_id)
) TABLESPACE pg_default;

create table public.messages (
  message_id uuid not null default gen_random_uuid(),
  chat_id uuid not null,
  sender_id uuid not null,
  sender_role text not null,
  content text not null,
  created_at timestamp without time zone null default now(),
  constraint messages_pkey primary key (message_id),
  constraint messages_chat_id_fkey foreign KEY (chat_id) references chats (chat_id) on update CASCADE on delete CASCADE,
  constraint messages_sender_id_fkey foreign KEY (sender_id) references auth.users (id) on update CASCADE on delete CASCADE,
  constraint messages_sender_role_check check (sender_role in ('customer', 'vendor'))
) TABLESPACE pg_default;

create table public.transactions (
  transaction_id uuid not null default gen_random_uuid(),
  customer_id uuid not null,
  vendor_id uuid not null,
  amount double precision not null,
  platform_fee double precision not null,
  status text not null default 'pending',
  created_at timestamp without time zone null default now(),
  constraint transactions_pkey primary key (transaction_id),
  constraint transactions_customer_id_fkey foreign KEY (customer_id) references customers (id) on update CASCADE on delete CASCADE,
  constraint transactions_vendor_id_fkey foreign KEY (vendor_id) references vendors (vendor_id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;
